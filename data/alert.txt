function validate() 
{
var name=document.reg.name.value;
  if(name==null || name == "" || name==" "|| name.length <3)
    {
    alert("Please Enter Name Correctly");
        return false;
    }
    
    var password = document.reg.password.value;
    if (password==null || password == "" || password==" "|| password.length <6) {
        alert("Enter password.. Minimum 6 characters");
        return false;
    }

var dob=document.reg.dob.value;
  if(dob==null || dob == "" || dob==" ")
    {
    alert("Enter Date Of Birth");
        return false;
    }
   var skill=document.reg.skill.value;
  if(skill==null || skill == "" || skill==" ")
    {
    alert("Enter skill");
        return false;
    }
    
    
    
    var blood_group=document.reg.blood_group.value;
  if(blood_group==null || blood_group == "" || blood_group==" ")
    {
    alert("Enter blood group");
        return false;
    }
    
    
    var mobile=document.reg.mobile.value;
  if(mobile==null || mobile == "" || mobile==" " ||mobile.length <10 ||mobile.length >10)
    {
    alert("Enter 10 digit number");
        return false;
    }
    
    
    var x = document.forms["reg"]["email"].value;
    var atpos = x.indexOf("@");
    var dotpos = x.lastIndexOf(".");
    if (atpos<1 || dotpos<atpos+2 || dotpos+2>=x.length) {
        alert("Not a valid e-mail address");
        return false;
    } 
  



var address=document.reg.address.value;
  if(address==null || address == "" || address==" ")
    {
    alert("Enter address");
        return false;
    }

}
